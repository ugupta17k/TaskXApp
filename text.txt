app.put("/notes", authmiddleware, async (req, res) => {
  const userId = req.userId;
  const noteid = req.query.noteid;
  const NewTitle = req.body.NewTitle;
  const NewDescription = req.body.NewDescription;

 
  const FindNote = await pool.query(`SELECT * FROM notes WHERE noteid = $1 AND userid = $2`, [noteid, userId]);

  if (FindNote.rows.length === 0) {
    return res.status(404).json({
      message: "Note not found or not owned by user"
    });
  }

  // Build dynamic update query based on provided fields
  let updateFields = [];
  let values = [];
  let paramIndex = 1;

  if (NewTitle !== undefined) {
    updateFields.push(`Title = $${paramIndex++}`);
    values.push(NewTitle);
  }
  if (NewDescription !== undefined) {
    updateFields.push(`description = $${paramIndex++}`);
    values.push(NewDescription);
  }

  if (updateFields.length === 0) {
    return res.status(400).json({
      message: "No fields to update"
    });
  }

  values.push(userId, noteid);

  const UpdateNote = await pool.query(
    `UPDATE notes SET ${updateFields.join(', ')} WHERE userid = $${paramIndex++} AND noteid = $${paramIndex} RETURNING *`,
    values 
  );

  res.json({
    message: "Note updated successfully",
    updatedNote: UpdateNote.rows[0]
  });
});
